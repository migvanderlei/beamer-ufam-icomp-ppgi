\begin{frame}{Etapas do Método}
  \begin{enumerate}
    \item \textbf{Preparação} — defina o conjunto inicial e os critérios
          de filtragem ou amostragem.

    \item \textbf{Representação} — construa protótipos, embeddings ou
          outra estrutura intermediária no espaço latente.

    \item \textbf{Refinamento} — aplique a regra de atualização
          (votação, limiar, otimização etc.).

    \item \textbf{Treinamento} — combine o subconjunto confiável com
          as instâncias refinadas e atualize o modelo.
  \end{enumerate}
  \vspace{0.4em}
  \begin{destaque}{Ideia-chave}
    Separe claramente o que controla a qualidade da âncora
    daquilo que define a cobertura do treinamento.
  \end{destaque}
\end{frame}
